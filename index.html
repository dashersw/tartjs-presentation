Modüller ve nesne yönelimli programlama


goog.provide('tart.MasterChef');
goog.require('tart.Chef');

/**
 * Master şef sınıfı tart'ın usta aşçılarının yaratıldığı sınıftır.
 * Onlar ki şüphesiz tart'ın en leziz yemeklerini yaparlar.
 *
 * @constructor
 * @extends {tart.Chef}
 * @param {string} name Şefin adı.
 * @param {string} profession Şefin ustalık alanı.
 */
tart.MasterChef = function(name, profession) {
    goog.base(this, name);
    this.profession = profession;
};
goog.inherits(tart.MasterChef, tart.Chef);

tart.MasterChef.prototype.greet = function() {
    var name = goog.base(this, 'greet');

    return 'Merhaba ben ' + name + ', ' + 
        this.profession + ' konusunda uzmanım.';
};

Uydurma syntax


$.Class('Monster',
/* @static */
{
  count: 0
},
/* @prototype */
{
  init: function( name ) {

    // saves name on the monster instance
    this.name = name;

    // sets the health
    this.health = 10;

    // increments count
    this.constructor.count++;
  },
  eat: function( smallChildren ){
    this.health += smallChildren;
  },
  fight: function() {
    this.health -= 2;
  }
});

/**
 * @constructor
 * @extends {tart.mvc.Controller}
 */
mvcapp.controllers.GamesController = function() {
    goog.base(this);
};
goog.inherits(mvcapp.controllers.GamesController, tart.mvc.Controller);

/**
 * @this {tart.mvc.Action}
 * @return {tart.mvc.uri.Redirection} Redirects to 'home anything' route.
 */
mvcapp.controllers.GamesController.indexAction = function() {
    this.setViewScript(mvcapp.views.scripts.games.index);
    this.setLayout(mvcapp.views.layouts.rare);

    this.view.p1 = this.params['p1'];
    this.view.p2 = this.params['p2'];

    if (this.view.p1 == 'a')
        return mvcapp.router.redirectToRoute('home anything',
            {'p1': 'pe1', 'p2': 'pe2', 'c': 'd'});
};
